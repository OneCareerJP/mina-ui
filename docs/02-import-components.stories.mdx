# Import components

## Fully import

```js
// main.js

import Vue from 'vue';
import MinaUi from 'mina-ui';
import 'mina/dist/mina-ui.css';

Vue.use(MinaUi);
```

## On demand

```js
// main.js

import Vue from 'vue';
import { MButton, MSelectbox } from 'mina-ui';
import 'mina/dist/mina-ui.css';

Vue.use(MButton);
Vue.use(MSelectbox);

new Vue({
  el: '#app',
  render: h => h(App)
});
```
